select trim(sm),
       dj,
       isbn,
       kc 库存,
       (select trim(mc) from t_bb where bh = s.bb) 版别,
       (select trim(mc) from t_ghdw where bh = s.ghdwh) 供货商,
       s.id
  from
       (select id, sum(kccs) kc from t_ltkf where hqbh = '13' group by id) t
       , t_kcsm s
 where s.id = t.id
