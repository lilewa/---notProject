where not exists (select 1 from  t_pkls_sl20171125 where hw=a.hw and id=a.id)

 
select    *  from t_pkls_sl20171125 

          update t_pkdz2  a
          set yscs=(select kccs from  T_LTKF_BAK where hw=a.hw and id=a.id )
          where exists (select 1 from  T_LTKF_BAK where hw=a.hw and id=a.id)
          
          update t_pkdz2 set yscs=0 where yscs is null 
        
  select trim(s.sm) 书名,
                 isbn,
                 dj 定价,
                 p.hw 货位,
                 pkcs 盘点数,
                 yscs 原册数,
                 q.mc 货区,
                 p.id,
                 q.bh
            from t_pkdz2 p, t_hqdy q, t_kcsm s
           where abs(pkcs - yscs) > 50
             and p.hqbh = q.bh
             and s.id = p.id


 --明细内容有：货区，品种，册数，码洋

  select q.mc,
         a.pz 立库品种,
         b.pz 盘点品种,
         a.cs 立库册数,
         b.cs 盘点册数,
         a.my 立库码洋,
         b.my 盘点码洋
    from (select count(distinct l.id) pz,
                 sum(kccs) cs,
                 sum(kccs * dj) my,
                 hqbh
            from t_ltkf l, t_kcsm s
           where l.id = s.id
           group by hqbh) a
         ,
         (select count(distinct l.id) pz,
                 sum(pkcs) cs,
                 sum(pkcs * dj) my,
                 hqbh
            from t_pkls_sl l, t_kcsm s
           where l.id = s.id
           group by hqbh) b,
         t_hqdy q
  where a.hqbh=b.hqbh and a.hqbh=q.bh
